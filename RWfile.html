// Java File Writing and Reading Example import java.io.*; public class FileReadWrite { public static void main(String[] args) { try { // Writing to file FileWriter writer = new FileWriter("output.txt"); writer.write("Hello, this is written to the file."); writer.close(); System.out.println("File written successfully."); // Reading from file BufferedReader reader = new BufferedReader(new FileReader("output.txt")); String line; while ((line = reader.readLine()) != null) { System.out.println("Read: " + line); } reader.close(); } catch (IOException e) { System.out.println("An error occurred."); e.printStackTrace(); } } }
